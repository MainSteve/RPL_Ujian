import javax.persistence.*;

@Entity
@Table(name = "doctor_schedule")
public class DoctorSchedule {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private int id;
    
    @Column(name = "doctor_name")
    private String doctorName;
    
    @Column(name = "specialist")
    private String specialist;
    
    @Column(name = "day")
    private String day;
    
    @Column(name = "time")
    private String time;

    // Getters dan Setters (asumsikan ada)
}

**2. Service: DoctorScheduleService.java**
```java
package com.mycompany.rpl_ujian.service;

import com.mycompany.rpl_ujian.model.DoctorSchedule;
import com.mycompany.rpl_ujian.repository.DoctorScheduleRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import java.util.List;
import java.util.Optional;

@Service
public class DoctorScheduleService {
    @Autowired
    private DoctorScheduleRepository repository;

    public DoctorSchedule save(DoctorSchedule schedule) {
        return repository.save(schedule);
    }

    public List<DoctorSchedule> findAll() {
        return repository.findAll();
    }

    public void delete(int id) {
        repository.deleteById(id);
    }
    
    public Optional<DoctorSchedule> findById(int id) {
        return repository.findById(id);
    }
}
```
*Penjelasan untuk AI*: Jelaskan bahwa Service ini menangani logika bisnis dan memanggil Repository. (Catatan: Jika proyek aslinya tidak pakai Spring, sesuaikan penjelasannya bahwa ini mengatur transaksi JPA).

**3. Controller: DoctorController.java**
```java
package com.mycompany.rpl_ujian.controller;

import com.mycompany.rpl_ujian.model.DoctorSchedule;
import com.mycompany.rpl_ujian.service.DoctorScheduleService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import java.util.List;

@Controller
public class DoctorController {
    @Autowired
    private DoctorScheduleService service;

    public void addSchedule(String name, String specialist, String day, String time) {
        DoctorSchedule schedule = new DoctorSchedule();
        schedule.setDoctorName(name);
        schedule.setSpecialist(specialist);
        schedule.setDay(day);
        schedule.setTime(time);
        service.save(schedule);
    }

    public List<DoctorSchedule> getAllSchedules() {
        return service.findAll();
    }

    public void deleteSchedule(int id) {
        service.delete(id);
    }
}
```
*Penjelasan untuk AI*: Jelaskan bahwa Controller menjadi penghubung antara View dan Service.
